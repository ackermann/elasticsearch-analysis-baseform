145    private AnalysisService createAnalysisService() {
146        Settings settings = ImmutableSettings.settingsBuilder()
147                .loadFromClasspath("org/xbib/elasticsearch/index/analysis/baseform_en.json").build();
148
149        Index index = new Index("test");
150
151        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
152                new EnvironmentModule(new Environment(settings)),
153                new IndicesAnalysisModule())
154                .createInjector();
155
156        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
157        new AnalysisBaseformPlugin().onModule(analysisModule);
158
159        Injector injector = new ModulesBuilder().add(
160                new IndexSettingsModule(index, settings),
161                new IndexNameModule(index),
162                analysisModule)
163                .createChildInjector(parentInjector);
164
165        return injector.getInstance(AnalysisService.class);
166    }
167
168    private void assertSimpleTSOutput(TokenStream stream, String[] expected) throws IOException {
169        stream.reset();
170        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
171        assertNotNull(termAttr);
172        int i = 0;
173        while (stream.incrementToken()) {
174            assertTrue(i < expected.length);
175            assertEquals(expected[i++], termAttr.toString(), "expected different term at index " + i);
176        }
177        assertEquals(i, expected.length, "not all tokens produced");
178        stream.close();
179    }
